// Tire Pressure Monitoring System (TPMS)
// Components: Arduino UNO, Pressure Sensor (MPX5700 or similar), Buzzer

#define PRESSURE_SENSOR A0
#define BUZZER 8

int pressureValue;
float voltage, pressure;

void setup() {
  pinMode(BUZZER, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  // Read analog sensor value
  pressureValue = analogRead(PRESSURE_SENSOR);

  // Convert to voltage (5V system, 10-bit ADC)
  voltage = (pressureValue / 1023.0) * 5.0;

  // Convert voltage to pressure (example formula for MPX5700)
  pressure = (voltage - 0.2) * (700.0 / 4.7); // in kPa

  Serial.print("Pressure: ");
  Serial.print(pressure);
  Serial.println(" kPa");

  // Alert if pressure drops below threshold (example: 200 kPa)
  if (pressure < 200) {
    digitalWrite(BUZZER, HIGH);
    Serial.println("⚠️ Low Tire Pressure Warning!");
  } else {
    digitalWrite(BUZZER, LOW);
  }

  delay(1000);
}
